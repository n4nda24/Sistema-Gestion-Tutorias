-- Creación de la base de datos
CREATE DATABASE IF NOT EXISTS gestion_tutorias;
USE gestion_tutorias;

-- 1. Tabla de Roles (Seguridad)
CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE -- ROLE_ADMIN, ROLE_DOCENTE, ROLE_ESTUDIANTE
);

-- 2. Tabla de Usuarios (Base para Herencia en Spring Boot)
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL, -- Almacenará el hash de BCrypt
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    tipo_usuario ENUM('ADMIN', 'DOCENTE', 'ESTUDIANTE') NOT NULL,
    activo BOOLEAN DEFAULT TRUE
);

-- 3. Tabla Intermedia para Roles (Many-to-Many)
CREATE TABLE usuario_roles (
    usuario_id INT NOT NULL,
    rol_id INT NOT NULL,
    PRIMARY KEY (usuario_id, rol_id),
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE,
    FOREIGN KEY (rol_id) REFERENCES roles(id) ON DELETE CASCADE
);

-- 4. Tabla de Asignaturas
CREATE TABLE asignaturas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    departamento VARCHAR(100)
);

-- 5. Tabla de Disponibilidad (Oferta del Docente)
CREATE TABLE disponibilidades (
    id INT AUTO_INCREMENT PRIMARY KEY,
    docente_id INT NOT NULL,
    asignatura_id INT NOT NULL,
    fecha DATE NOT NULL,
    hora_inicio TIME NOT NULL,
    hora_fin TIME NOT NULL,
    ocupado BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (docente_id) REFERENCES usuarios(id),
    FOREIGN KEY (asignatura_id) REFERENCES asignaturas(id)
);

-- 6. Tabla de Tutorías (Reservas)
CREATE TABLE tutorias_agendadas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    disponibilidad_id INT NOT NULL UNIQUE,
    estudiante_id INT NOT NULL,
    fecha_reserva TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado ENUM('PENDIENTE', 'REALIZADA', 'CANCELADA') DEFAULT 'PENDIENTE',
    observaciones TEXT,
    FOREIGN KEY (disponibilidad_id) REFERENCES disponibilidades(id),
    FOREIGN KEY (estudiante_id) REFERENCES usuarios(id)
);
